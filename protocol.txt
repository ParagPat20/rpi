   INIT : 1
   CLOSE : 1

   DISARM : 1

   NED : (X,Y,Z,T)
   YAW : (H,R)

   MTL : (DIS, ALT, NEWS_DEG)
   RCOV : (STATE, CH1, CH2, CH3, CH4, CH5, CH6)

   GET_MODE : (MODES)
   SPEED : AIRSPD, GNDSPD, VELOCITY

   IS_CONNECTED : 1 OR 0
